
func main(harvestOverride)
// FUNCTION BEGIN
  autoPickup()
  autoHarvest(harvestOverride)

  return
// FUNCTION END




func autoPickup()
// FUNCTION BEGIN
  var pickupRange

  ? pickup = treasure
    pickupRange = 99
  :
    pickupRange = 2

  ? pickup.distance <= pickupRange
    equipL Triskelion Stone
    equipR Star Stone

  return
// FUNCTION END




func autoHarvest(harvestOverride)
// FUNCTION BEGIN
  ? harvestOverride = 1
    return


  var harvestTool = ""

  ? loc.name = Rocky Plateau
    harvestTool = "Shovel"
  :?loc.name = Deadwood Canyon
    harvestTool = "Hatchet"
  :
    return


  var harvestRange = 6
  ? harvest.distance <= harvestRange
    equipL Triskelion Stone

    ? harvestTool = Hatchet
      equipR @harvestTool@
    :?harvestTool = Shovel
      equip  @harvestTool@


  ? item.right = Hatchet
^ & item.right.state > 1
^ & item.GetCooldown("hatchet") < 1
    activate R

  return
// FUNCTION END
